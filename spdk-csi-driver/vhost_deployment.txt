# Dockerfile for SPDK CSI Driver with vhost support
apiVersion: v1
kind: ConfigMap
metadata:
  name: spdk-csi-dockerfile
  namespace: spdk-system
data:
  Dockerfile: |
    FROM ubuntu:22.04

    # Install dependencies
    RUN apt-get update && apt-get install -y \
        build-essential \
        pkg-config \
        libnuma-dev \
        libssl-dev \
        libaio-dev \
        libcunit1-dev \
        libncurses5-dev \
        uuid-dev \
        python3 \
        python3-pip \
        git \
        curl \
        qemu-utils \
        qemu-system-x86 \
        libvirt-clients \
        libvirt-daemon-system \
        hugepages \
        nvme-cli \
        util-linux \
        parted \
        udev \
        && rm -rf /var/lib/apt/lists/*

    # Install Rust
    RUN curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh -s -- -y
    ENV PATH="/root/.cargo/bin:${PATH}"

    # Install vhost-user-blk
    RUN git clone https://github.com/rust-vmm/vhost-device.git /tmp/vhost-device && \
        cd /tmp/vhost-device && \
        cargo build --release --bin vhost-user-blk && \
        cp target/release/vhost-user-blk /usr/local/bin/ && \
        rm -rf /tmp/vhost-device

    # Build SPDK with vhost support
    RUN git clone https://github.com/spdk/spdk --recurse-submodules /opt/spdk && \
        cd /opt/spdk && \
        git checkout v24.01 && \
        ./configure --with-vhost --with-virtio --enable-debug && \
        make -j$(nproc)

    # Set up SPDK environment
    ENV SPDK_PATH=/opt/spdk
    ENV PATH="${SPDK_PATH}/build/bin:${PATH}"

    # Copy CSI driver binary (built separately)
    COPY spdk-csi-driver /usr/local/bin/
    COPY spdk-csi-controller /usr/local/bin/
    COPY spdk-csi-node-agent /usr/local/bin/

    # Set up entrypoint
    COPY entrypoint.sh /entrypoint.sh
    RUN chmod +x /entrypoint.sh

    ENTRYPOINT ["/entrypoint.sh"]

---
apiVersion: v1
kind: ConfigMap
metadata:
  name: spdk-csi-entrypoint
  namespace: spdk-system
data:
  entrypoint.sh: |
    #!/bin/bash
    set -e

    # Set up hugepages
    echo 2048 > /sys/kernel/mm/hugepages/hugepages-2048kB/nr_hugepages

    # Bind NVMe devices to vfio-pci if specified
    if [ -n "$NVME_DEVICES" ]; then
        cd /opt/spdk
        for device in $(echo $NVME_DEVICES | tr ',' ' '); do
            echo "Binding $device to vfio-pci"
            scripts/setup.sh config $device
        done
    fi

    # Start SPDK target with vhost support
    if [ "$1" = "spdk-target" ]; then
        exec /opt/spdk/build/bin/spdk_tgt \
            -m 0x3 \
            -p 0 \
            -s 512 \
            --wait-for-rpc
    fi

    # Start CSI driver components
    exec "$@"

apiVersion: v1
kind: ConfigMap
metadata:
  name: spdk-vhost-config
  namespace: spdk-system
data:
  setup-vhost.sh: |
    #!/bin/bash
    # Setup script for vhost environment
    
    # Configure hugepages
    echo 2048 > /proc/sys/vm/nr_hugepages
    
    # Mount hugepages if not already mounted
    if ! mountpoint -q /dev/hugepages; then
        mount -t hugetlbfs nodev /dev/hugepages
    fi
    
    # Load vfio modules
    modprobe vfio-pci
    modprobe vhost-net
    modprobe vhost-scsi
    
    # Setup IOMMU
    echo 1 > /sys/module/vfio/parameters/enable_unsafe_noiommu_mode
    
    # Create socket directory
    mkdir -p /var/lib/spdk-csi/sockets
    chmod 755 /var/lib/spdk-csi/sockets
