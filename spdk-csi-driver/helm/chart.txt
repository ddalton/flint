apiVersion: apps/v1
kind: Deployment
metadata:
  name: {{ .Chart.Name }}-controller
  labels:
    {{- include "spdk-csi-driver.labels" . | nindent 4 }}
spec:
  replicas: {{ .Values.controller.replicas }}
  selector:
    matchLabels:
      {{- include "spdk-csi-driver.selectorLabels" . | nindent 6 }}
      app: spdk-csi-controller

        {{- include "spdk-csi-driver.selectorLabels" . | nindent 8 }}

      securityContext:

        fsGroup: {{ .Values.csiDriver.security.runAsGroup }}
      containers:
      - name: csi-provisioner
        image: {{ .Values.csiDriver.provisionerImage.repository }}:{{ .Values.csiDriver.provisionerImage.tag }}
        args:
        - "--csi-address=$(ADDRESS)"
        - "--v=2"
        - "--feature-gates=Topology=true"
        - "--enable-leader-election"
        - "--leader-election-type=leases"
        - "--timeout=300s"
        env:
        - name: ADDRESS
          value: {{ .Values.csiDriver.endpoint }}
        volumeMounts:
        - name: socket-dir
          mountPath: /var/lib/csi/sockets/pluginproxy
      - name: csi-resizer
        image: {{ .Values.csiDriver.resizerImage.repository }}:{{ .Values.csiDriver.resizerImage.tag }}
        args:
        - "--csi-address=$(ADDRESS)"
        - "--v=2"
        - "--timeout=300s"
        - "--handle-volume-inuse-error=false"
        env:
        - name: ADDRESS
          value: {{ .Values.csiDriver.endpoint }}
        volumeMounts:
        - name: socket-dir
          mountPath: /var/lib/csi/sockets/pluginproxy
      - name: csi-snapshotter
        image: {{ .Values.csiDriver.snapshotterImage.repository }}:{{ .Values.csiDriver.snapshotterImage.tag }}
        args:
        - "--csi-address=$(ADDRESS)"
        - "--v=2"
        - "--timeout=300s"
        - "--enable-distributed-snapshotting=false"
        env:
        - name: ADDRESS
          value: {{ .Values.csiDriver.endpoint }}
        volumeMounts:
        - name: socket-dir
          mountPath: /var/lib/csi/sockets/pluginproxy
      - name: csi-driver
        image: {{ .Values.csiDriver.image.repository }}:{{ .Values.csiDriver.image.tag }}
        imagePullPolicy: {{ .Values.csiDriver.image.pullPolicy }}
        env:
        - name: NODE_ID
          value: "controller"
        - name: CSI_ENDPOINT
          value: {{ .Values.csiDriver.endpoint }}
        - name: SPDK_RPC_SOCKET  # Changed from SPDK_RPC_URL
          value: {{ .Values.spdk.rpc.socketPath | quote }}
        securityContext:
          privileged: true
          capabilities:
            add: ["SYS_ADMIN"]
            drop: ["ALL"]
          allowPrivilegeEscalation: true
        volumeMounts:
        - name: socket-dir
          mountPath: /var/lib/csi/sockets/pluginproxy
        - name: spdk-socket
          mountPath: /var/run/spdk
      volumes:
      - name: socket-dir
        emptyDir: {}
      - name: spdk-socket
        hostPath:
          path: /var/run/spdk
